Step 4: Import Necessary Libraries

import plotly
import pandas as pd
import plotly.express as px

Step 5: Load the Dataset

ReadMe = pd.read_csv('/content/average_monthly_temperature_by_state_1950-2022.csv')
ReadMe = ReadMe.sort_values('year', ascending=True)
print(ReadMe)

Step 6: Calculate Yearly Average Temperature for Each State

yearly_avg_temp = ReadMe.groupby(['year', 'state'])['average_temp'].mean().reset_index()
yearly_avg_temp = yearly_avg_temp.sort_values(by=['year', 'average_temp'], ascending=[True, False])

Step 7: Create a Bar Plot

fig = px.bar(
    yearly_avg_temp,
    x='average_temp',
    y='state',
    animation_frame='year',
    animation_group='state',
    color='average_temp',
    hover_name='state',
    orientation='h',
    range_x=[yearly_avg_temp['average_temp'].min(), yearly_avg_temp['average_temp'].max()]
)

fig.update_layout(
    height=1200,
    paper_bgcolor='black',
    plot_bgcolor='black',
    font_color='white',
    yaxis=dict(
        tickfont=dict(color='white'),
        automargin=True,
        categoryorder='total ascending'
    ),
    xaxis=dict(
        tickfont=dict(color='white'),
        title='Average Temperature'
    ),
    title='Average Temperature of US States Over Time',
    title_font=dict(size=24),
    margin=dict(l=20, r=20, t=40, b=20),
    updatemenus=[{
        'buttons': [
        {
            'args': [None, {'frame': {'duration': 2200, 'redraw': True}, 'fromcurrent': True, 'transition': {'duration': 2200}}], #Change duration based on your preference
            'label': 'Play',
            'method': 'animate'
        },
        {
            'args': [[None], {'frame': {'duration': 0, 'redraw': True}, 'mode': 'immediate', 'transition': {'duration': 0}}],
            'label': 'Pause',
            'method': 'animate'
        }
    ],
        'direction': 'left',
        'pad': {'r': 10, 't': 87},
        'showactive': False,
        'type': 'buttons',
        'x': 0.1,
        'xanchor': 'right',
        'y': 0,
        'yanchor': 'top'
    }]
)

fig.show()

Final Visualization

This is an interactive bar plot showing the average temperatures of US states over time.
